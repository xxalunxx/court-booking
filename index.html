// Supabase Edge Function: paymongo-webhook // Endpoint registered in PayMongo Dashboard -> Developers -> Webhooks // Subscribed events (per your current setup): // payment.paid, qr.paid -> mark booking as paid // qr.expired, qrph.expired -> mark booking as expired (frees the slot) // source.chargeable -> informational only, logged but no action // (Payment Intent capture_type "automatic" // already handles the charge - no need to // manually create a Payment like the older // Sources-only flow required) // // IMPORTANT: this function must have "Enforce JWT Verification" turned OFF // in its Settings tab, since PayMongo calls it without a Supabase auth token. // // Required secrets (project-wide, should already be available): // PAYMONGO_SECRET_KEY <- test secret key, used for the metadata lookup fallback // PAYMONGO_WEBHOOK_SECRET <- shown in PayMongo's webhook details (for future signature verification) // // NOTE: exact payload shape for qr.paid / qr.expired / qrph.expired / source.chargeable // hasn't been fully confirmed against live payloads yet - resolveRefCode tries several // common paths defensively. If any of these don't match a booking, check the // payment_events table's raw_payload column to see the real shape and we can // tighten the lookup. import { createClient } from "https://esm.sh/@supabase/supabase-js@2"; async function resolveRefCode(resource: any): Promise { // 1. Direct metadata on the resource itself (works for payment.paid) let refCode = resource?.attributes?.metadata?.ref_code; if (refCode) return refCode; // 2. Sometimes nested one level deeper under a linked source/intent refCode = resource?.attributes?.source?.attributes?.metadata?.ref_code; if (refCode) return refCode; // 3. Fall back to fetching the linked Payment Intent from PayMongo's API const paymentIntentId = resource?.attributes?.payment_intent_id || resource?.attributes?.payment_intent || (resource?.type === "payment_intent" ? resource?.id : null) || resource?.attributes?.source?.id || resource?.attributes?.source_id || resource?.id; // last resort: maybe the resource itself IS the intent/source id if (!paymentIntentId) return null; try { const secretKey = Deno.env.get("PAYMONGO_SECRET_KEY")!; const auth = "Basic " + btoa(`${secretKey}:`); const piRes = await fetch(`https://api.paymongo.com/v1/payment_intents/${paymentIntentId}`, { headers: { Authorization: auth }, }); const piData = await piRes.json(); return piData?.data?.attributes?.metadata?.ref_code || null; } catch { return null; } } async function updateBookingStatus( supabase: any, refCode: string, newStatus: "paid" | "expired", source: string, rawPayload: string, ) { const updateFields: Record = { status: newStatus }; if (newStatus === "paid") updateFields.paid_at = new Date().toISOString(); const { data: updated, error } = await supabase .from("bookings") .update(updateFields) .eq("ref_code", refCode) .eq("status", "held") // only move rows that are still held - safe to call twice for paid+qr.paid both firing .select(); await supabase.from("payment_events").insert({ source, raw_payload: rawPayload, detected_ref_code: refCode, matched: !!(updated && updated.length), matched_booking_id: updated && updated.length ? updated[0].id : null, }); return { updated, error }; } async function logOnly(supabase: any, refCode: string | null, source: string, rawPayload: string) { await supabase.from("payment_events").insert({ source, raw_payload: rawPayload, detected_ref_code: refCode, matched: false, }); } const PAID_EVENTS = ["payment.paid", "qr.paid"]; const EXPIRED_EVENTS = ["qr.expired", "qrph.expired"]; Deno.serve(async (req) => { try { const rawBody = await req.text(); const event = JSON.parse(rawBody); const supabase = createClient( Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!, ); const eventType = event.data?.attributes?.type; const resource = event.data?.attributes?.data; if (PAID_EVENTS.includes(eventType)) { const refCode = await resolveRefCode(resource); if (!refCode) { await logOnly(supabase, null, "paymongo_webhook", rawBody); return new Response(JSON.stringify({ received: true, warning: "No ref_code found for " + eventType }), { status: 200 }); } const { updated, error } = await updateBookingStatus(supabase, refCode, "paid", "paymongo_webhook", rawBody); if (error) return new Response(JSON.stringify({ error: error.message }), { status: 500 }); return new Response(JSON.stringify({ received: true, event: eventType, updated: updated?.length || 0 }), { status: 200 }); } if (EXPIRED_EVENTS.includes(eventType)) { const refCode = await resolveRefCode(resource); if (!refCode) { await logOnly(supabase, null, "paymongo_webhook", rawBody); return new Response(JSON.stringify({ received: true, warning: "No ref_code found for " + eventType }), { status: 200 }); } const { updated, error } = await updateBookingStatus(supabase, refCode, "expired", "paymongo_webhook", rawBody); if (error) return new Response(JSON.stringify({ error: error.message }), { status: 500 }); return new Response(JSON.stringify({ received: true, event: eventType, expired: updated?.length || 0 }), { status: 200 }); } if (eventType === "source.chargeable") { // Informational only - Payment Intent auto-capture handles the charge. const refCode = await resolveRefCode(resource); await logOnly(supabase, refCode, "paymongo_webhook", rawBody); return new Response(JSON.stringify({ received: true, event: eventType, action: "logged_only" }), { status: 200 }); } return new Response(JSON.stringify({ received: true, ignored: eventType }), { status: 200 }); } catch (e) { return new Response(JSON.stringify({ error: String(e) }), { status: 500 }); } });
